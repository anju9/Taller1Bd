create tablespace gofar_travel datafile 
'gofar_traveler_1.dbf' size 500M,
'gofar_traveler_2.dbf' size 400M,
'gofar_traveler_3.dbf' size 100M

ONLINE;

'test_purposes_1.dbf' size 500M
